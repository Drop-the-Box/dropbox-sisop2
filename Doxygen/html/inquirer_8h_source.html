105 void ask()
106 {
107 if (!m_title.empty())
108 std::cout << "\033[34m>\033[0m " << m_title << '\n';
109 for (auto& question : m_questions) {
110 auto& q = question.second;
111
112 auto printQuestion = [&](const std::string& append = "") {
113 std::cout << "\033[1m\033[34m?\033[0m \033[1m" << q.m_question << "\033[0m " << append;
114 };
115
116 auto takeInput = [](std::string& destination) {
117 std::cout << "\033[34m";
118 std::getline(std::cin, destination);
119 std::cout << "\033[0m";
120 };
121
122 switch (q.m_type) {
123 case Type::confirm: {
124 printQuestion("(y/N) ");
125 std::string answer;
126 takeInput(answer);
127 while (!(answer == "y" || answer == "Y" || answer == "n" || answer == "N")) {
128 erase_lines(2);
129 printQuestion("(y/N) ");
130 takeInput(answer);
131 }
132 q.m_answer = answer;
133 break;
134 }
135 case Type::text:
136 printQuestion();
137 takeInput(q.m_answer);
138 break;
139 case Type::integer: {
140 printQuestion();
141 std::string answer;
142 takeInput(answer);
143 while (!is_integer(answer)) {
144 erase_lines(2);
145 printQuestion();
146 takeInput(answer);
147 }
148 q.m_answer = answer;
149 break;
150 }
151 case Type::decimal: {
152 printQuestion();
153 std::string answer;
154 takeInput(answer);
155 while (!is_decimal(answer)) {
156 erase_lines(2);
157 printQuestion();
158 takeInput(answer);
159 }
160 q.m_answer = answer;
161 break;
162 }
163 case Type::yesNo: {
164 const std::string yes = "\033[34myes\033[0m no\n";
165 const std::string no = "yes \033[34mno\033[0m\n";
166 std::cout << std::flush;
167 printQuestion(yes);
168 bool position = true;
169 while (true) {
170 const int key = getch();
171 if (key == Inquirer::keySx) {
172 position = true;
173 erase_lines(2);
174 printQuestion(yes);
175 }
176 else if (key == Inquirer::keyDx) {
177 position = false;
178 erase_lines(2);
179 printQuestion(no);
180 }
181 if (key == Inquirer::keyEnter) {
182 q.m_answer = position ? "yes" : "no";
183 break;
184 }
185 }
186 break;
187 }
188 case Type::options: {
189 unsigned int selectedIndex = 0;
190 auto printOptions = [&]() {
191 std::cout << '\n';
192 for (int i = 0; i < q.m_options.size(); ++i) {
193 if (i == selectedIndex)
194 std::cout << "\033[34m> " << q.m_options.at(i) << "\033[0m\n";
195 else
196 std::cout << " " << q.m_options.at(i) << "\n";
197 }
198 };
199 printQuestion();
200 printOptions();
201
202 while (true) {
203 const int key = getch();
204 if (key == Inquirer::keyDw) {
205 selectedIndex = wrap_int(selectedIndex + 1, 0, q.m_options.size() - 1);
206 erase_lines(q.m_options.size() + 2);
207 printQuestion();
208 printOptions();
209 }
210 else if (key == Inquirer::keyUp) {
211 selectedIndex = wrap_int(selectedIndex - 1, 0, q.m_options.size() - 1);
212 erase_lines(q.m_options.size() + 2);
213 printQuestion();
214 printOptions();
215 }
216 if (key == Inquirer::keyEnter) {
217 q.m_answer = q.m_options.at(selectedIndex);
218 erase_lines(q.m_options.size() + 2);
219 printQuestion("\033[34m" + q.m_options.at(selectedIndex) + "\033[0m\n");
220 break;
221 }
222 }
223 break;
224 }
225 case Type::regex: {
226 printQuestion();
227 std::string answer;
228 takeInput(answer);
229 while (!std::regex_match(answer, std::regex(q.m_regex))) {
230 erase_lines(2);
231 printQuestion();
232 takeInput(answer);
233 }
234 q.m_answer = answer;
235 break;
236 }
237 }
238 }

277 static void erase_lines(const unsigned count = 1)
278 {
279 if (count == 0)
280 return;

282 std::cout << "\x1b[2K"; // Delete current line

290 static bool is_integer(const std::string& string)
291 {
292 if (string.empty() || ((!isdigit(string[0])) && (string[0] != '-') && (string[0] != '+'))) return false;
293 char* p;
294 strtol(string.c_str(), &p, 10);
295 return (*p == 0);
296 }
297
298 static bool is_decimal(const std::string& string)
299 {
300 if (string.empty() || ((!isdigit(string[0])) && (string[0] != '-') && (string[0] != '+'))) return false;
301 char* p;
302 strtod(string.c_str(), &p);
303 return (*p == 0);
304 }

306 static unsigned wrap_int(unsigned int k, const unsigned lowerBound, const unsigned upperBound)
307 {
308 const unsigned range_size = upperBound - lowerBound + 1;
309
310 if (k < lowerBound)
311 k += range_size * ((lowerBound - k) / range_size + 1);
312
313 return lowerBound + (k - lowerBound) % range_size;
314 }

316 static int getch()
317 {
318 int c; // This function should return the keystroke without allowing it to echo on screen
319
320 #ifdef _WIN32
321 c = _getch();
322 #else
323 system("stty raw"); // Raw input - wait for only a single keystroke
324 system("stty -echo"); // Echo off
325 c = getchar();
326 system("stty cooked"); // Cooked input - reset
327 system("stty echo"); // Echo on - Reset
328 #endif
329
330 return c;
