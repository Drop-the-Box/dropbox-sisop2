93 PLOGI << "get_message_sync returned: " << msg.get() << endl;
94 sleep(4);
95 unique_ptr<Packet> resp_packet(new Packet(msg.get()));
96 if (resp_packet->type == EventMsg) {

102 PLOGD << "Event detail: " << evt->message << endl;
