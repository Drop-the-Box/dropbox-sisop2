194 {
195 va_list ap_copy;
196#if defined(_MSC_VER) && _MSC_VER <= 1600
197 ap_copy = ap; // there is no va_copy on Visual Studio 2010
198#else
199 va_copy(ap_copy, ap);
200#endif
201#ifndef __STDC_SECURE_LIB__
202 int charCount = vsnprintf(NULL, 0, format, ap_copy);
203#else
204 int charCount = _vscprintf(format, ap_copy);
205#endif
206 va_end(ap_copy);
207 if (charCount < 0)
208 {
209 return -1;
210 }
211
212 size_t bufferCharCount = static_cast<size_t>(charCount) + 1;

215 if (!str)

219
220#ifndef __STDC_SECURE_LIB__
221 int retval = vsnprintf(str, bufferCharCount, format, ap);
222#else

224#endif
225 if (retval < 0)
226 {
227 free(str);
228 return -1;
229 }
230
231 *strp = str;

336 inline std::string processFuncName(const char* func)
337 {
338#if (defined(_WIN32) && !defined(__MINGW32__)) || defined(__OBJC__)
339 return std::string(func);
340#else
341 const char* funcBegin = func;
342 const char* funcEnd = ::strchr(funcBegin, '(');
343 int foundTemplate = 0;
344
345 if (!funcEnd)
346 {
347 return std::string(func);
348 }
349
350 for (const char* i = funcEnd - 1; i >= funcBegin; --i) // search backwards for the first space char
351 {
352 if (*i == '>')
353 {
354 foundTemplate++;
355 }
356 else if (*i == '<')
357 {
358 foundTemplate--;
359 }
360 else if (*i == ' ' && foundTemplate == 0)
361 {
362 funcBegin = i + 1;
363 break;
364 }
365 }
366
367 return std::string(funcBegin, funcEnd);
368#endif
369 }
